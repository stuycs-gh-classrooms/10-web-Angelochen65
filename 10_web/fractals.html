This is my modified koch curve. I added a second turtle to mirror the pattern made by the first turtle. I also made it so each koch curve of the koch curve is a random color. Finally, I updated the base case of the original koch curve to this:

This is my updated koch_curve with a depth of 3:

 
turt = turtle.Turtle()
turt1 = turtle.Turtle()
def mirror_koch(t, t1, depth, length):
    if depth == 1:
        t.pencolor(random.randrange(255),random.randrange(255),random.randrange(255))
        t.fd(length)
        t1.pencolor(random.randrange(255),random.randrange(255),random.randrange(255))
        t1.fd(length)
        t.lt(60)
        t1.rt(60)
        t.pencolor(random.randrange(255),random.randrange(255),random.randrange(255))
        t.fd(length)
        t1.pencolor(random.randrange(255),random.randrange(255),random.randrange(255))
        t1.fd(length)
        t.rt(120)
        t1.lt(120)
        t.pencolor(random.randrange(255),random.randrange(255),random.randrange(255))
        t.fd(2 * length)
        t1.pencolor(random.randrange(255),random.randrange(255),random.randrange(255))
        t1.fd(2 * length)
        t.lt(120)
        t1.rt(120)
        t.pencolor(random.randrange(255),random.randrange(255),random.randrange(255))
        t.fd(length)
        t1.pencolor(random.randrange(255),random.randrange(255),random.randrange(255))
        t1.fd(length)
        t.rt(60)
        t1.lt(60)
        t.pencolor(random.randrange(255),random.randrange(255),random.randrange(255))
        t.fd(length)
        t1.pencolor(random.randrange(255),random.randrange(255),random.randrange(255))
        t1.fd(length)
    else:
        mirror_koch(t, t1, depth -1, length)
        t.lt(60)
        t1.rt(60)
        mirror_koch(t, t1, depth -1, length)
        t.rt(120)
        t1.lt(120)
        mirror_koch(t, t1, depth -1, length)
        t.lt(60)
        t1.rt(60)
        mirror_koch(t,t1, depth -1, length)
        
mirror_koch(turt,turt1,3,10)

Sierpinski Triangle

For the sierpinski triangle, I modified the base case so that the second side of the triangle is not drawn, and the first side of the triangle is a thicker gray line so the entire figure looks more like a shadow-y arrow rather than a triangle, though you can still see the triangles.

Original Sierpinski with depth of 5:

My variation with depth of 4:

        
def draw_sierpinski(t, depth, length):
    if depth == 1:
        t.lt(60)
        t.fd(length)
        t.rt(120)
        t.pencolor('white')
        t.pensize(1)
        t.fd(length)
        t.rt(120)
        t.pencolor('black')
        t.fd(length)
        t.rt(180)
        t.pencolor('gray')
        t.pensize(4)
    else:
        draw_sierpinski(t, depth - 1, length/2)
        t.fd(length/2)
        draw_sierpinski(t, depth - 1, length/2)
        t.fd(-length/2)
        t.lt(60)
        t.fd(length / 2)
        t.rt(60)
        draw_sierpinski(t, depth - 1, length/2)
        t.rt(120)
        t.fd(length/2)
        t.lt(120)

draw_sierpinski(turt,4,200)

Fractal Tree

For my version of the fractal tree, I added color so that the trunk and branches would be brown while the leaves would be green. I also changed the initial tree from being a line, to being rectangles, which have randomized dimensions.

Original tree fractal with a depth of 4:

My tree with a depth of 4:

        
            
turt.lt(90)
def tree(t, depth, length, angle):
    if depth == 1:
        t.pencolor('green')
        t.fd(length)
        t.rt(90)
        a = random.randrange(10)
        while a == 0:
            a = random.randrange(10)
        t.fd(length/a)
        t.rt(90)
        t.fd(length)
        t.rt(90)
        t.fd(length/a)
        t.rt(90)
    else:
        t.pencolor('brown')
        t.fd(length)
        t.rt(90)
        a = random.randrange(10)
        while a == 0:
            a = random.randrange(10)
        t.fd(length/a)
        t.rt(90)
        t.fd(length)
        t.rt(90)
        t.fd(length/a)
        t.rt(90)
        t.pensize(2)
        t.fd(length)
        t.lt(angle)
        tree(t, depth - 1, length, angle)
        t.rt(4 * angle)
        t.fd(length/a)
        t.lt(angle)
        tree(t,depth -1,length,angle)
        t.rt(angle)
        t.fd(-length/a)
        t.lt(90)
        t.fd(-length)    

tree(turt,6,50,30)
